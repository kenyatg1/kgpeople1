People 1.py 
## List all people
db.people.find().pretty()

## Count all people
db.people.countDocuments()

## List all people in Arizona
db.people.find({ state: "Arizona" }).pretty()

## List all males in Arizona
db.people.find({ state: "Arizona", gender: "male" }).pretty()

## List all people in Arizona plus New Mexico
db.people.find({ state: { $in: ["Arizona", "New Mexico"] } }).pretty()

## List all people under age 40
db.people.find({ age: { $lt: 40 } }).pretty()

## List all females in Florida between the ages of 40 and 45 (inclusive)
db.people.find({ state: "Florida", gender: "female", age: { $gte: 40, $lte: 45 } }).pretty()

## List people whose first name starts with "H"
db.people.find({ firstName: { $regex: /^H/, $options: 'i' } }).pretty()

## List all people in Michigan, sorted by first name
db.people.find({ state: "Michigan" }).sort({ firstName: 1 }).pretty()

## List all people who live in Virginia or are named Virginia
db.people.find({ $or: [{ state: "Virginia" }, { firstName: "Virginia" }] }).pretty()

## List the names of people under age 30. Only display their first and last name
db.people.find({ age: { $lt: 30 } }, { firstName: 1, lastName: 1, _id: 0 }).pretty()

## List all people in Montana. Display all information except age
db.people.find({ state: "Montana" }, { age: 0 }).pretty()

## List the email addresses of people with a ".edu" email. Only display the email
db.people.find({ email: { $regex: /\.edu$/ } }, { email: 1, _id: 0 }).pretty()
